<Steps>

  </Step>
  <Step title="Contact Us">
    Email [support@indefi.io](mailto:support@indefi.io) with subject line: "Partner Verification Request"
  </Step>
  <Step title="Provide Context">
    Include your organization name and the specific information you're seeking
  </Step>
</Steps>
